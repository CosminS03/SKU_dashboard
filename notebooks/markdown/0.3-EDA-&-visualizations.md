```python
def calc_units(df: pd.DataFrame, year: int, sold: bool):
    if sold:
        return (
            df[df["InvoiceDate"].dt.year == year][~df["InvoiceNo"].str.startswith("C")]
            .groupby(by="SKU")["Quantity"]
            .sum()
            .reset_index()
        )
    else:
        return (
            df[df["InvoiceDate"].dt.year == year][df["InvoiceNo"].str.startswith("C")]
            .groupby(by="SKU")["Quantity"]
            .sum()
            .abs()
            .reset_index()
        )
```
