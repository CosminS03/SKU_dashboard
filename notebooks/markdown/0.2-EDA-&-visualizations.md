```python
import pandas as pd

df = pd.read_csv("../data/interim/online_retail_transformed.csv")
```


```python
import matplotlib.pyplot as plt
import seaborn as sns
```


```python
df["Revenue"] = round(df["Quantity"] * df["UnitPrice"], 2)

df_rev = df[~df["InvoiceNo"].str.startswith('C')].groupby("SKU")["Revenue"].sum().reset_index()

df_rev = df_rev.sort_values(by="Revenue", ascending=False).reset_index()

df_rev["Cummulutive"] = df_rev["Revenue"].cumsum()
total_sum = df_rev["Revenue"].sum()
df_rev["Cummulutive Perc"] = round(df_rev["Cummulutive"] / total_sum * 100)

sku_order = df_rev["SKU"].to_list()
```


```python
fig, ax1 = plt.subplots()

sns.barplot(x = "SKU", y="Revenue", data=df_rev, ax=ax1, color="blue", order=sku_order)
ax1.set_ylabel = "Frequency"
ax1.set_xticks([])

ax2 = ax1.twinx()
ax2.plot(df_rev.index, df_rev["Cummulutive Perc"], color = "red", marker = "o", linestyle = "--")

ax2.set_ylabel = "Cummulutive %"
ax2.set_ylim(0, 110)

plt.title("Revenue Pareto chart")
plt.show()
```
